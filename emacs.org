* Backups
#+BEGIN_SRC emacs-lisp
  (setq backup-directory-alist
        '(("." . "~/.saves")))
  (setq delete-old-versions t
        kept-new-versions 6
        kept-old-versions 2
        version-control t)
  (setq backup-by-copying t) ; safest, but slowest
#+END_SRC
