*** Generalized list of programming hooks
    Depends on the list of lisp hooks to already be defined.
#+BEGIN_SRC emacs-lisp
  (defvar *programming-hooks*
    (append
     ,*lisp-hooks*
     '(ruby-mode-hook
       js2-mode-hook
       js-mode-hook
       c-mode-hook
       shell-script-mode-hook
       c++-mode-hook
